8from typing import get_args, get_origin

10from .custom_serializers import DEFAULT_CUSTOM_SERIALIZERS 

11from .types import PYTHON_TYPE_TO_AVRO_MAPPING, SIMPLE_PYTHON_TYPE_TO_AVRO_MAPPING 

12 

13DEFAULT_FIELD_TYPES = {serializer.cls: serializer.field_type for serializer in DEFAULT_CUSTOM_SERIALIZERS} 

14 

15 

16def schema_from_signal(signal, custom_type_to_avro_type=None): 

17 """ 

18 Create an Avro schema for events sent by an instance of OpenEdxPublicSignal. 

19 

20 Arguments: 

21 signal: An instance of OpenEdxPublicSignal 

22 custom_type_to_avro_type: A map of Python class to Avro type 

23 """ 

24 field_types = custom_type_to_avro_type or {} 

25 all_custom_field_types = {**DEFAULT_FIELD_TYPES, **field_types} 

26 previously_seen_types = set() 

27 

28 base_schema = { 

29 "name": "CloudEvent", 

30 "type": "record", 

31 "doc": "Avro Event Format for CloudEvents created with openedx_events/schema", 

32 "fields": [], 

33 "namespace": signal.event_type, 

34 } 

35 

36 for data_key, data_type in signal.init_data.items(): 

37 base_schema["fields"].append(_create_avro_field_definition(data_key, data_type, 

38 previously_seen_types, 

39 custom_type_to_avro_type=all_custom_field_types)) 

40 return base_schema

43def _create_avro_field_definition(data_key, data_type, previously_seen_types, 

44 custom_type_to_avro_type=None, default_is_none=False): 

45 """ 

46 Create an Avro schema field definition from an OpenEdxPublicSignal data definition. 

47 

48 Arguments: 

49 data_key: Field name, eg `lms_user_id`, `course_id`, `enrollment_status` 

50 data_type: Python data type, eg `str`, `CourseKey`, `CourseEnrollmentData` 

51 previously_seen_types: list of previously-encountered data types 

52 custom_type_to_avro_type: map of custom data types to a pre-determined avro field type 

53 default_is_none: boolean indicating whether this field has 'None' as a default 

54 """ 

55 field = {"name": data_key} 

56 all_field_type_overrides = custom_type_to_avro_type or {} 

57 # get generic type of data_type 

58 # if data_type == List[int], data_type_origin = list 

59 data_type_origin = get_origin(data_type) 

60 

61 # Case 1: data_type has a predetermined avro field representation 

62 if field_type := all_field_type_overrides.get(data_type, None): 

63 field["type"] = field_type 

64 # Case 2: data_type is a simple type that can be converted directly to an Avro type 

65 elif data_type in PYTHON_TYPE_TO_AVRO_MAPPING: 

66 if PYTHON_TYPE_TO_AVRO_MAPPING[data_type] in ["record", "array"]: 

67 # pylint: disable-next=broad-exception-raised 

68 raise Exception("Unable to generate Avro schema for dict or array fields without annotation types.") 

69 avro_type = PYTHON_TYPE_TO_AVRO_MAPPING[data_type] 

70 field["type"] = avro_type 

71 elif data_type_origin == list: 

72 # returns types of list contents 

73 # if data_type == List[int], arg_data_type = (int,) 

74 arg_data_type = get_args(data_type) 

75 if not arg_data_type: 

76 raise TypeError( 

77 "List without annotation type is not supported. The argument should be a type, for eg., List[int]" 

78 ) 

79 avro_type = SIMPLE_PYTHON_TYPE_TO_AVRO_MAPPING.get(arg_data_type[0]) 

80 if avro_type is None: 

81 raise TypeError( 

82 "Only following types are supported for list arguments:" 

83 f" {set(SIMPLE_PYTHON_TYPE_TO_AVRO_MAPPING.keys())}" 

84 ) 

85 field["type"] = {"type": PYTHON_TYPE_TO_AVRO_MAPPING[data_type_origin], "items": avro_type} 

86 # Case 3: data_type is an attrs class 

87 elif hasattr(data_type, "__attrs_attrs__"): 

88 # Inner Attrs Class 

89 

90 # fastavro does not allow you to redefine the same record type more than once, 

91 # so only define an attr record once 

92 if data_type.__name__ in previously_seen_types: 

93 field["type"] = data_type.__name__ 

94 else: 

95 previously_seen_types.add(data_type.__name__) 

96 record_type = {"name": data_type.__name__, "type": 'record', "fields": []} 

97 

98 for attribute in data_type.__attrs_attrs__: 

99 record_type["fields"].append( 

100 _create_avro_field_definition(attribute.name, attribute.type, 

101 previously_seen_types, 

102 custom_type_to_avro_type=all_field_type_overrides, 

103 default_is_none=attribute.default is None) 

104 ) 

105 field["type"] = record_type 

106 else: 

107 raise TypeError( 

108 f"Data type {data_type} is not supported. The data type needs to either" 

109 " be one of the types in PYTHON_TYPE_TO_AVRO_MAPPING, an attrs decorated class, or one of the types" 

110 " defined in custom_type_to_avro_type." 

111 ) 

112 if default_is_none: 

113 field["default"] = None 

114 single_type = field["type"] 

115 field["type"] = ["null", single_type] 

116 return field
